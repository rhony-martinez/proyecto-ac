19 * - Sistema de actuadores: servomotor (persianas), ventilador DC, buzzer (alarma), LEDs RGB
20 * - Interfaz visual con LCD 16x2 para feedback al usuario
21 * - Máquina de estados robusta con 7 estados y transiciones asíncronas
22 * - Sistema de identificación por tarjetas RFID con almacenamiento en EEPROM
23 * - Detección de condiciones críticas y sistema de alarma con sensor de movimiento

30 * - PMV_BAJO: Activación de sistemas de calefacción (PMV < -1)

406/**
407 * @brief Configura todas las transiciones y callbacks de la FSM
408 *
409 * @details
410 * Define la lógica completa del sistema mediante:
411 * - Transiciones entre estados basadas en condiciones de entrada
412 * - Callbacks de entrada (OnEntering) para inicialización de estados
413 * - Callbacks de salida (OnLeaving) para limpieza de recursos
414 * - Funciones lambda para evaluación de condiciones en tiempo de ejecución
415 *
416 * @dot
417 * digraph FSM {
418 * rankdir=TB;
419 * node [shape=ellipse, style=filled, fillcolor=lightblue];
420 * edge [color=darkgreen, fontsize=10];
421 *
422 * INICIO [fillcolor=gold];
423 * BLOQUEO [fillcolor=red];
424 * ALARMA [fillcolor=orange];
425 * PMV_ALTO [fillcolor=lightcoral];
426 * PMV_BAJO [fillcolor=lightcyan];
427 * CONFIG [fillcolor=lightgreen];
428 * MONITOR [fillcolor=lightyellow];
429 *
430 * INICIO -> CONFIG [label="CLAVE_CORRECTA"];
431 * INICIO -> BLOQUEO [label="SISTEMA_BLOQUEADO"];
432 * BLOQUEO -> INICIO [label="TECLA_ASTERISCO"];
433 * CONFIG -> MONITOR [label="TIEMPO_EXPIRADO"];
434 * MONITOR -> PMV_BAJO [label="PMV_MENOR_QUE_MENOS1"];
435 * MONITOR -> PMV_ALTO [label="PMV_MAYOR_QUE_1"];
436 * MONITOR -> CONFIG [label="TIEMPO_EXPIRADO"];
437 * PMV_BAJO -> MONITOR [label="TIEMPO_EXPIRADO"];
438 * PMV_ALTO -> MONITOR [label="TIEMPO_EXPIRADO"];
439 * PMV_ALTO -> ALARMA [label="TEMP_ALTA_3_INTENTOS"];
440 * ALARMA -> INICIO [label="SENSOR_INFRARROJO"];
441 * }
442 * @enddot
443 *
444 * @note Cada transición es evaluada en cada ciclo del loop principal
445 * @see State
446 * @see Input
447 */
448void setupStateMachine() {
449 // ==================== TRANSICIONES DESDE INICIO ====================
450 stateMachine.AddTransition(INICIO, CONFIG, []() {
451 return input == CLAVE_CORRECTA;
452 });
453 stateMachine.AddTransition(INICIO, BLOQUEO, []() {
454 return input == SISTEMA_BLOQUEADO;
455 });
456
457 // ==================== TRANSICIONES DESDE BLOQUEO ====================
458 stateMachine.AddTransition(BLOQUEO, INICIO, []() {
459 return input == TECLA_ASTERISCO;
460 });
461
462 // ==================== TRANSICIONES DESDE CONFIG ====================
463 stateMachine.AddTransition(CONFIG, MONITOR, []() {
464 return input == TIEMPO_EXPIRADO;
465 });
466
467 // ==================== TRANSICIONES DESDE MONITOR ====================
468 stateMachine.AddTransition(MONITOR, PMV_BAJO, []() {
469 return input == PMV_MENOR_QUE_MENOS1;
470 });
471 stateMachine.AddTransition(MONITOR, PMV_ALTO, []() {
472 return input == PMV_MAYOR_QUE_1;
473 });
474 stateMachine.AddTransition(MONITOR, CONFIG, []() {
475 return input == TIEMPO_EXPIRADO;
476 });
477
478 // ==================== TRANSICIONES DESDE ESTADOS PMV ====================
479 stateMachine.AddTransition(PMV_BAJO, MONITOR, []() {
480 return input == TIEMPO_EXPIRADO;
481 });
482 stateMachine.AddTransition(PMV_ALTO, MONITOR, []() {
483 return input == TIEMPO_EXPIRADO;
484 });
485 stateMachine.AddTransition(PMV_ALTO, ALARMA, []() {
486 return input == TEMP_ALTA_3_INTENTOS;
487 });
488
489 // ==================== TRANSICIONES DESDE ALARMA ====================
490 stateMachine.AddTransition(ALARMA, INICIO, []() {
491 return input == SENSOR_INFRARROJO;
492 });
493
494 // ==================== CONFIGURACIÓN DE CALLBACKS DE ENTRADA ====================
495 stateMachine.SetOnEntering(INICIO, outputInicio);
496 stateMachine.SetOnEntering(BLOQUEO, outputBloqueo);
497 stateMachine.SetOnEntering(CONFIG, outputConfig);
498 stateMachine.SetOnEntering(MONITOR, outputMonitor);
499 stateMachine.SetOnEntering(PMV_BAJO, outputPMV_Bajo);
500 stateMachine.SetOnEntering(PMV_ALTO, outputPMV_Alto);
501 stateMachine.SetOnEntering(ALARMA, outputAlarma);
502
503 // ==================== CONFIGURACIÓN DE CALLBACKS DE SALIDA ====================
504 stateMachine.SetOnLeaving(BLOQUEO, onLeavingBloqueo);
505 stateMachine.SetOnLeaving(ALARMA, []() {
506 TaskBuzzer.Stop();
507 digitalWrite(BUZZER_PIN, LOW);
508 });
509 stateMachine.SetOnLeaving(PMV_BAJO, onLeavingPMV_Bajo);
510 stateMachine.SetOnLeaving(PMV_ALTO, onLeavingPMV_Alto);
511}

518 * @brief Función de inicialización del sistema Arduino
519 *
520 * @details
521 * Configuración inicial que se ejecuta una vez al iniciar el sistema:
522 * - Inicializa comunicación serial a 9600 baudios para monitoreo y debugging
523 * - Configura display LCD 16x2 con pines específicos
524 * - Inicializa sensor DHT11 para lectura de temperatura y humedad
525 * - Configura servomotor en pin PWM específico
526 * - Establece estado inicial de la FSM en INICIO
527 * - Configura pines de LEDs RGB, buzzer, ventilador y sensor PIR como salidas
528 * - Inicializa sistema SPI y módulo RFID MFRC522
529 * - Configura sistema de tareas asíncronas
530 *
531 * @pre Todos los componentes deben estar correctamente conectados según diagrama
532 * @post Sistema listo para ejecutar loop principal y procesamiento FSM
533 *
534 * @see setupStateMachine()
535 * @see loop()
536 */

538 Serial.begin(9600);
539 lcd.begin(16, 2);
540 dht.begin();
541 servo.attach(SERVO_PIN);
542
543 // Configuración de pines de salida
544 pinMode(LED_RED, OUTPUT);
545 pinMode(LED_GREEN, OUTPUT);
546 pinMode(LED_BLUE, OUTPUT);
547 pinMode(BUZZER_PIN, OUTPUT);
548 pinMode(FAN_PIN, OUTPUT);

1104 *

1112 }

1116/**
1117 * @brief Busca un UID en la memoria EEPROM
1118 *
1119 * @param uid Array de bytes con el UID a buscar
1120 * @param uidLength Longitud del UID
1121 * @return int Dirección en EEPROM si se encuentra, -1 si no existe
1122 *
1123 * @details
1124 * Recorre todos los registros en EEPROM comparando el UID
1125 * proporcionado con los almacenados. Considera tanto la
1126 * longitud como cada byte del UID.
1127 *
1128 * @see MAX_RECORDS
1129 * @see RECORD_SIZE
1130 */
1131int findUIDInEEPROM(byte* uid, byte uidLength) {
1132 for (int recordIndex = 0; recordIndex < MAX_RECORDS; recordIndex++) {
1133 int address = recordIndex * RECORD_SIZE;
1134 byte storedLen = EEPROM.read(address);
1135
1136 // Registro vacío → saltar
1137 if (storedLen == 0xFF) continue;
1138
1139 // Si la longitud no coincide → no es la misma tarjeta
1140 if (storedLen != uidLength) continue;
1141
1142 // Comparar byte a byte
1143 bool match = true;
1144 for (int i = 0; i < uidLength; i++) {
1145 byte storedByte = EEPROM.read(address + 1 + i);
1146 if (storedByte != uid[i]) {
1147 match = false;
1148 break;
1149 }
1150 }
1151
1152 if (match) {
1153 return address; // encontrado
1154 }
1155 }
1156 return -1; // no encontrado
1157}

1159/**
1160 * @brief Lee el nombre asociado a un registro desde EEPROM
1161 *
1162 * @param recordAddress Dirección base del registro en EEPROM
1163 * @param buffer Buffer donde almacenar el nombre leído
1164 * @param maxLen Longitud máxima del buffer
1165 *
1166 * @details
1167 * Extrae el campo de nombre de un registro RFID almacenado
1168 * en EEPROM. El nombre se almacena después del UID en la
1169 * estructura del registro.
1170 */
1171void readNameFromEEPROM(int recordAddress, char* buffer, int maxLen) {
1172 int base = recordAddress;
1173 for (int i = 0; i < maxLen; i++) {
1174 buffer[i] = EEPROM.read(base + 1 + MAX_UID_LENGTH + i);
1175 if (buffer[i] == '\0') break;
1176 }
1177}

1179/**
1180 * @brief Guarda un nuevo UID y nombre en EEPROM
1181 *
1182 * @param uid Array de bytes con el UID de la tarjeta
1183 * @param uidLength Longitud del UID
1184 * @param name Nombre asociado a la tarjeta
1185 *
1186 * @details
1187 * Busca la primera posición libre en EEPROM y almacena:
1188 * - Longitud del UID (1 byte)
1189 * - UID completo (hasta MAX_UID_LENGTH bytes)
1190 * - Nombre (hasta NAME_MAX_LENGTH caracteres)
1191 *
1192 * @security Los registros vacíos se identifican por 0xFF
1193 * @see MAX_RECORDS
1194 * @see RECORD_SIZE
1195 */
1196void saveUIDAndName(byte* uid, byte uidLength, const char* name) {
1197 for (int recordIndex = 0; recordIndex < MAX_RECORDS; recordIndex++) {
1198 int address = recordIndex * RECORD_SIZE;
1199 byte firstByte = EEPROM.read(address);
1200
1201 if (firstByte == 0xFF) { // posición libre (EEPROM limpia es 0xFF)
1202 // Guardar longitud UID
1203 EEPROM.write(address, uidLength);
1204
1205 // Guardar UID
1206 for (int i = 0; i < uidLength; i++) {
1207 EEPROM.write(address + 1 + i, uid[i]);
1208 }
1209
1210 // Guardar nombre
1211 for (int i = 0; i < NAME_MAX_LENGTH; i++) {
1212 if (i < strlen(name)) {
1213 EEPROM.write(address + 1 + MAX_UID_LENGTH + i, name[i]);
1214 } else {
1215 EEPROM.write(address + 1 + MAX_UID_LENGTH + i, '\0');
1216 }
1217 }
1218
1219 Serial.print(" Guardado en registro ");
1220 Serial.println(recordIndex);
1221 break;
1222 }
1223 }
1224}

1582/**
1583 * @brief Calcula el índice PMV (Predicted Mean Vote) usando modelo de Fanger
1584 *
1585 * @param ta Temperatura del aire [°C]
1586 * @param tr Temperatura radiante media [°C]
1587 * @param rh Humedad relativa [%]
1588 * @param vel_ar Velocidad del aire [m/s]
1589 * @param M Tasa metabólica [W/m²]
1590 * @param clo Aislamiento térmico de la ropa [clo]
1591 * @return float Índice PMV en escala de -3 a +3
1592 *
1593 * @details
1594 * Implementa el modelo completo de Fanger para cálculo de confort térmico:
1595 * - -3: Muy frío | -2: Frío | -1: Ligeramente frío
1596 * - 0: Neutral (confort óptimo)
1597 * - +1: Ligeramente cálido | +2: Cálido | +3: Muy cálido
1598 *
1599 * @algorithm
1600 * 1. Conversión de unidades y constantes base
1601 * 2. Iteración para hallar temperatura de superficie de ropa (tcl)
1602 * 3. Cálculo de pérdidas de calor (radiación, convección, respiración)
1603 * 4. Cálculo final del índice PMV
1604 *
1605 * @equation PMV = (0.303 × e^{-0.036M} + 0.028) × (M - W - H)
1606 *
1607 * @warning Resultado limitado al rango [-3, +3] según estándar ISO 7730
1608 *
1609 * @reference Fanger, P. O. (1970). Thermal Comfort. Danish Technical Press.
1610 * @reference ISO 7730:2005 - Ergonomics of the thermal environment
1611 */
1612float calcularPMV_Fanger(float ta, float tr, float rh, float vel_ar, float M, float clo) {
1613 // Conversión de unidades y constantes base
1614 float pa, icl, V, fcl, hc, pmv;
1615 float tol = 0.0001; // Tolerancia para iteraciones
1616 float tcl = ta; // Temperatura inicial de la superficie de la ropa
1617 float tcl_prev;
1618 int iteracion = 0;
1619 int max_iters = 100;
1620
1621 // Constantes del modelo
1622 V = 0.0; // Trabajo mecánico (reposo) [W/m²]
1623 icl = clo * 0.155; // Conversión clo a m²K/W
1624 pa = calcularPresionVapor(ta, rh); // Presión de vapor [Pa]
1625
1626 // Factor de superficie de la ropa
1627 fcl = calcularFcl(icl);
1628
1629 // ------------ Iteración para hallar tcl ------------
1630 do {
1631 tcl_prev = tcl;
1632
1633 // Calcular coeficiente de convección
1634 hc = calcularHc(tcl, ta, vel_ar);
1635
1636 // Conversión a Kelvin para fórmulas radiativas
1637 float tr_rad = tr + 273.0;
1638 float tcl_rad = tcl + 273.0;
1639
1640 // Cálculo de transferencia de calor radiante
1641 float radiation = (3.96 * pow(10, -8)) * fcl * (pow(tcl_rad, 4) - pow(tr_rad, 4));
1642 // Cálculo de transferencia de calor por convección
1643 float convection = fcl * hc * (tcl - ta);
1644
1645 // Ecuación de balance térmico para tcl
1646 tcl = 35.7 - 0.028 * (M - V) - icl * (radiation - convection);
1647
1648 iteracion++;
1649 } while (fabs(tcl - tcl_prev) > tol && iteracion < max_iters);
1650
1651 // Cálculo final de coeficiente de convección
1652 float hc_final = calcularHc(tcl, ta, vel_ar);
1653
1654 // -------------- CÁLCULO FINAL DEL PMV --------------
1655 // Conversión a Kelvin
1656 float tr_rad = tr + 273.0;
1657 float tcl_rad = tcl + 273.0;
1658
1659 // Cálculo de todas las pérdidas de calor
1660 float radiation = (3.96e-8) * fcl * (pow(tcl_rad, 4) - pow(tr_rad, 4));
1661 float convection = fcl * hc_final * (tcl - ta);
1662 float respiration = 0.0014 * M * (34.0 - ta);
1663 float latent = (1.7e-5) * M * (5867.0 - pa);
1664 float heat_loss = latent + respiration + radiation + convection;
1665
1666 // Ecuación final del PMV
1667 pmv = (0.303 * exp(-0.036 * M) + 0.028) * (
1668 (M - V)
1669 - (3.05e-3) * (5733.0 - 6.99 * (M - V) - pa)
1670 - 0.42 * ((M - V) - 58.15)
1671 - heat_loss
1672 );
1673
1674 // Limitar resultado al rango estándar [-3, +3]
1675 return constrain(pmv, -3.0, 3.0);
1676}
